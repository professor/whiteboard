def self.index_tank

  @index ||= @api.indexes(ENV['SEARCHIFY_INDEX'] || 'cmux')
  @index
end

def self.search(query)
  index_tank.search("#{query} OR title:#{query}", :fetch => 'timestamp,url,text,title', :snippet => 'text')
